Employee Class


#include <iostream>
using namespace std;

class employee {
public:
    string Name;
    string Company;
    int Age;

    employee(string name, string company, int age) {
        Name = name;
        Company = company;
        Age = age;
    }

    void introduce() {
        cout << Name << " works at " << Company << " and is " << Age << " years old" << endl;
    }
};

int main() {
    employee e1("Khaled", "Amazon", 16);
    e1.introduce();
}
Demonstrates basic class structure, object creation, and constructor usage in C++.

Mobile (Encapsulation)


#include <iostream>
using namespace std;

class mobile {
private:
    string Brand;
    string Model;
    int Price;

public:
    void setbrand(string brand) {
        Brand = brand;
    }

    void setmodel(string model) {
        Model = model;
    }

    void setprice(int price) {
        if (price >= 0)
            Price = price;
    }

    void show() {
        cout << Brand << " " << Model << " | $" << Price << endl;
    }

    mobile(string brand, string model, int price) {
        setbrand(brand);
        setmodel(model);
        setprice(price);
    }
};

int main() {
    mobile m("Samsung", "S23", 2000);
    m.show();
}
Applies encapsulation by protecting data and validating input using setters.

User Account (Validation)


#include <iostream>
using namespace std;

class useraccount {
private:
    string Username;
    string Password;
    int Age;

public:
    useraccount(string username, string password, int age) {
        if (password.length() >= 6 && age >= 13) {
            Username = username;
            Password = password;
            Age = age;
        }
    }

    void show() {
        cout << Username << " | Age: " << Age << endl;
    }
};

int main() {
    useraccount u("Khaled", "1234567", 19);
    u.show();
}
Implements real-world validation rules for age and password security.

Movie Class (Final)


#include <iostream>
using namespace std;

class movie {
private:
    string Title;
    string Director;
    int Rating;

public:
    movie(string title, string director, int rating) {
        if (rating >= 1 && rating <= 10) {
            Title = title;
            Director = director;
            Rating = rating;
        }
    }

    void show() {
        cout << Title << " - " << Director << " | Rating: " << Rating << endl;
    }
};

int main() {
    movie m("Inception", "Nolan", 9);
    m.show();
}
Shows clean encapsulation with defensive programming to prevent invalid data.
